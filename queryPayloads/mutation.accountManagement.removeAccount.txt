mutation accountManagementRemoveAccount ( $accountId:ID! ) {
	accountManagement ( accountId:$accountId ) {
		removeAccount  {
			accountInfoRemoveAccountPayload: accountInfo {
				id 
				name 
				description 
				timeZone 
				type 
				tenancy 
				audit  {
					createdBy
					createdTime
				}

				plan 
				status 
			}
		}

	}	
}
