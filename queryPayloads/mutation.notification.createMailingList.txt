mutation notificationCreateMailingList ( $accountId:ID! $createMailingListInput:CreateMailingListInput!) {
	notification ( accountId:$accountId ) {
		createMailingList ( input:$createMailingListInput  )  {
			mailingList {
				id
				name
				type
				address
				admin {
					id
					name
				}
				audit {
					updatedTime
					updatedBy {
						id
						name
					}
				}
			}
		}
	}	
}
