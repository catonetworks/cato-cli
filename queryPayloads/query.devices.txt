query devices ( $deviceV2Input:DeviceV2Input $sortOrderInput:SortOrderInput $deviceAttributeCatalogInput:DeviceAttributeCatalogInput $deviceCsvExportInput:DeviceCsvExportInput $jobId:ID! $accountId:ID! ) {
	devices ( accountId:$accountId ) {
		list ( input:$deviceV2Input  )  {
			deviceDevicesPayload: device {
				id 
				site  {
					id
					name
				}

				user  {
					id
					name
				}

				name 
				firstSeen 
				lastSeen 
				ip 
				ipAddress 
				category 
				riskScore 
				nic  {
					macAddress
					vendor
				}

				os  {
					product
					vendor
					version
				}

				hw  {
					type
					model
					manufacturer
				}

				originTypes 
				isManaged 
				network  {
					subnet
					networkName
				}

				networkInfo 
				connectionProfile  {
					destApps
					destDomains
					destHosts
					directions
				}

				confidence 
			}
			pagingDevicesPayload: paging {
				total 
			}
		}

		attributesCatalog  {
			categoryDeviceAttributesCatalogQueries: category ( sort:$sortOrderInput   ) {
				items 
				pageInfo  {
					total
				}

			}
			typeDeviceAttributesCatalogQueries: type ( input:$deviceAttributeCatalogInput   ) {
				items 
				pageInfo  {
					total
				}

			}
			modelDeviceAttributesCatalogQueries: model ( input:$deviceAttributeCatalogInput   ) {
				items 
				pageInfo  {
					total
				}

			}
			manufacturerDeviceAttributesCatalogQueries: manufacturer ( input:$deviceAttributeCatalogInput   ) {
				items 
				pageInfo  {
					total
				}

			}
			osDeviceAttributesCatalogQueries: os ( input:$deviceAttributeCatalogInput   ) {
				items 
				pageInfo  {
					total
				}

			}
		}

		csvExport ( input:$deviceCsvExportInput  )  {
			jobId
			message
		}

		csvExportStatus ( jobId:$jobId  )  {
			jobId
			statusExportStatusResponse: status
			progress
			message
			downloadUrl
			expiresAt
		}

	}	
}
