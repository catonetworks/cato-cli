query devices ( $deviceV2Input:DeviceV2Input $sortOrderInput:SortOrderInput $deviceAttributeCatalogInput:DeviceAttributeCatalogInput $jobId:ID! ) {
	devices ( ) {
		list ( input:$deviceV2Input  )  {
			deviceDevicesPayload: device {
				id
				siteDeviceV2: site {
					id
					name
				}

				userDeviceV2: user {
					id
					name
				}

				name
				firstSeen
				lastSeen
				ip
				category
				riskScore
				nicDeviceV2: nic {
					macAddress
					vendor
				}

				osDeviceV2: os {
					product
					vendor
					version
				}

				hwDeviceV2: hw {
					type
					model
					manufacturer
				}

				networkDeviceV2: network {
					subnet
					networkName
				}

				networkInfoDeviceV2: networkInfo
				connectionProfileDeviceV2: connectionProfile {
					destApps
					destDomains
					destHosts
					directions
				}

				confidenceDeviceV2: confidence
			}
			pagingDevicesPayload: paging {
				total
			}
		}

		attributesCatalog {
			categoryDeviceAttributesCatalogQueries: category ( sort:$sortOrderInput   ) {
				items
				pageInfoDeviceAttributeCatalogPayload: pageInfo {
					total
				}

			}
			typeDeviceAttributesCatalogQueries: type ( input:$deviceAttributeCatalogInput   ) {
				items
				pageInfoDeviceAttributeCatalogPayload: pageInfo {
					total
				}

			}
			modelDeviceAttributesCatalogQueries: model ( input:$deviceAttributeCatalogInput   ) {
				items
				pageInfoDeviceAttributeCatalogPayload: pageInfo {
					total
				}

			}
			manufacturerDeviceAttributesCatalogQueries: manufacturer ( input:$deviceAttributeCatalogInput   ) {
				items
				pageInfoDeviceAttributeCatalogPayload: pageInfo {
					total
				}

			}
			osDeviceAttributesCatalogQueries: os ( input:$deviceAttributeCatalogInput   ) {
				items
				pageInfoDeviceAttributeCatalogPayload: pageInfo {
					total
				}

			}
		}

		csvExport {
			jobId
			message
		}

		csvExportStatus ( jobId:$jobId  )  {
			jobId
			statusExportStatusResponse: status
			progress
			message
			downloadUrl
			expiresAt
		}

	}	
}
