	groups ( accountId:$accountId ) {
		whereUsed ( input:$groupRefInput  )  {
			used
			usageGroupWhereUsedPayload: usage {
				type 
				field 
			}
			invalidMemberTypesGroupWhereUsedPayload: invalidMemberTypes {
				type 
				scope  {
					type
					field
				}

			}
		}

	}	
}
