query popLocations ( $accountId:ID! $popLocationFilterInput:PopLocationFilterInput $popLocationAllocatedIpInput:PopLocationAllocatedIpInput) {
	popLocations ( accountId:$accountId ) {
		popLocationList ( input:$popLocationFilterInput  )  {
			items {
				id
				name
				displayName
				country {
					id
					name
				}
				isPrivate
				cloudInterconnect {
					taggingMethod
					providerName
				}
			}
		}
		allocatedIpList ( input:$popLocationAllocatedIpInput  )  {
			allocatedIp {
				id
				account {
					id
					name
				}
				ip
				popLocation {
					id
					name
				}
				allocationType
				description
				subnet
				state
			}
			paging {
				total
			}
		}
	}	
}
